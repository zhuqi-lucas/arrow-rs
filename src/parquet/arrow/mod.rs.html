18//! API for reading/writing Arrow [`RecordBatch`]es and [`Array`]s to/from
19//! Parquet Files.
20//!
21//! See the [crate-level documentation](crate) for more details on other APIs
22//!
23//! # Schema Conversion
24//!
25//! These APIs ensure that data in Arrow [`RecordBatch`]es written to Parquet are
26//! read back as [`RecordBatch`]es with the exact same types and values.
27//!
28//! Parquet and Arrow have different type systems, and there is not
29//! always a one to one mapping between the systems. For example, data
30//! stored as a Parquet [`BYTE_ARRAY`] can be read as either an Arrow
31//! [`BinaryViewArray`] or [`BinaryArray`].
32//!
33//! To recover the original Arrow types, the writers in this module add a "hint" to
34//! the metadata in the [`ARROW_SCHEMA_META_KEY`] key which records the original Arrow
35//! schema. The metadata hint follows the same convention as arrow-cpp based
36//! implementations such as `pyarrow`. The reader looks for the schema hint in the
37//! metadata to determine Arrow types, and if it is not present, infers the Arrow schema
38//! from the Parquet schema.
39//!
40//! In situations where the embedded Arrow schema is not compatible with the Parquet
41//! schema, the Parquet schema takes precedence and no error is raised.
42//! See [#1663](https://github.com/apache/arrow-rs/issues/1663)
43//!
44//! You can also control the type conversion process in more detail using:
45//!
46//! * [`ArrowSchemaConverter`] control the conversion of Arrow types to Parquet
47//!   types.
48//!
49//! * [`ArrowReaderOptions::with_schema`] to explicitly specify your own Arrow schema hint
50//!   to use when reading Parquet, overriding any metadata that may be present.
51//!
52//! [`RecordBatch`]: arrow_array::RecordBatch
53//! [`Array`]: arrow_array::Array
54//! [`BYTE_ARRAY`]: crate::basic::Type::BYTE_ARRAY
55//! [`BinaryViewArray`]: arrow_array::BinaryViewArray
56//! [`BinaryArray`]: arrow_array::BinaryArray
57//! [`ArrowReaderOptions::with_schema`]: arrow_reader::ArrowReaderOptions::with_schema
58//!
59//! # Example: Writing Arrow `RecordBatch` to Parquet file
60//!
61//!```rust
62//! # use arrow_array::{Int32Array, ArrayRef};
63//! # use arrow_array::RecordBatch;
64//! # use parquet::arrow::arrow_writer::ArrowWriter;
65//! # use parquet::file::properties::WriterProperties;
66//! # use tempfile::tempfile;
67//! # use std::sync::Arc;
68//! # use parquet::basic::Compression;
69//! let ids = Int32Array::from(vec![1, 2, 3, 4]);
70//! let vals = Int32Array::from(vec![5, 6, 7, 8]);
71//! let batch = RecordBatch::try_from_iter(vec![
72//!   ("id", Arc::new(ids) as ArrayRef),
73//!   ("val", Arc::new(vals) as ArrayRef),
74//! ]).unwrap();
75//!
76//! let file = tempfile().unwrap();
77//!
78//! // WriterProperties can be used to set Parquet file options
79//! let props = WriterProperties::builder()
80//!     .set_compression(Compression::SNAPPY)
81//!     .build();
82//!
83//! let mut writer = ArrowWriter::try_new(file, batch.schema(), Some(props)).unwrap();
84//!
85//! writer.write(&batch).expect("Writing batch");
86//!
87//! // writer must be closed to write footer
88//! writer.close().unwrap();
89//! ```
90//!
91//! # Example: Reading Parquet file into Arrow `RecordBatch`
92//!
93//! ```rust
94//! # use std::fs::File;
95//! # use parquet::arrow::arrow_reader::ParquetRecordBatchReaderBuilder;
96//! # use std::sync::Arc;
97//! # use arrow_array::Int32Array;
98//! # use arrow::datatypes::{DataType, Field, Schema};
99//! # use arrow_array::RecordBatch;
100//! # use parquet::arrow::arrow_writer::ArrowWriter;
101//! #
102//! # let ids = Int32Array::from(vec![1, 2, 3, 4]);
103//! # let schema = Arc::new(Schema::new(vec![
104//! #     Field::new("id", DataType::Int32, false),
105//! # ]));
106//! #
107//! # let file = File::create("data.parquet").unwrap();
108//! #
109//! # let batch = RecordBatch::try_new(Arc::clone(&schema), vec![Arc::new(ids)]).unwrap();
110//! # let batches = vec![batch];
111//! #
112//! # let mut writer = ArrowWriter::try_new(file, Arc::clone(&schema), None).unwrap();
113//! #
114//! # for batch in batches {
115//! #     writer.write(&batch).expect("Writing batch");
116//! # }
117//! # writer.close().unwrap();
118//! #
119//! let file = File::open("data.parquet").unwrap();
120//!
121//! let builder = ParquetRecordBatchReaderBuilder::try_new(file).unwrap();
122//! println!("Converted arrow schema is: {}", builder.schema());
123//!
124//! let mut reader = builder.build().unwrap();
125//!
126//! let record_batch = reader.next().unwrap().unwrap();
127//!
128//! println!("Read {} records.", record_batch.num_rows());
129//! ```
130//!
131//! # Example: Reading non-uniformly encrypted parquet file into arrow record batch
132//!
133//! Note: This requires the experimental `encryption` feature to be enabled at compile time.
134//!
135#![cfg_attr(feature = "encryption", doc = "```rust")]
136#![cfg_attr(not(feature = "encryption"), doc = "```ignore")]
137//! # use arrow_array::{Int32Array, ArrayRef};
138//! # use arrow_array::{types, RecordBatch};
139//! # use parquet::arrow::arrow_reader::{
140//! #     ArrowReaderMetadata, ArrowReaderOptions, ParquetRecordBatchReaderBuilder,
141//! # };
142//! # use arrow_array::cast::AsArray;
143//! # use parquet::file::metadata::ParquetMetaData;
144//! # use tempfile::tempfile;
145//! # use std::fs::File;
146//! # use parquet::encryption::decrypt::FileDecryptionProperties;
147//! # let test_data = arrow::util::test_util::parquet_test_data();
148//! # let path = format!("{test_data}/encrypt_columns_and_footer.parquet.encrypted");
149//! #
150//! let file = File::open(path).unwrap();
151//!
152//! // Define the AES encryption keys required required for decrypting the footer metadata
153//! // and column-specific data. If only a footer key is used then it is assumed that the
154//! // file uses uniform encryption and all columns are encrypted with the footer key.
155//! // If any column keys are specified, other columns without a key provided are assumed
156//! // to be unencrypted
157//! let footer_key = "0123456789012345".as_bytes(); // Keys are 128 bits (16 bytes)
158//! let column_1_key = "1234567890123450".as_bytes();
159//! let column_2_key = "1234567890123451".as_bytes();
160//!
161//! let decryption_properties = FileDecryptionProperties::builder(footer_key.to_vec())
162//!     .with_column_key("double_field", column_1_key.to_vec())
163//!     .with_column_key("float_field", column_2_key.to_vec())
164//!     .build()
165//!     .unwrap();
166//!
167//! let options = ArrowReaderOptions::default()
168//!  .with_file_decryption_properties(decryption_properties);
169//! let reader_metadata = ArrowReaderMetadata::load(&file, options.clone()).unwrap();
170//! let file_metadata = reader_metadata.metadata().file_metadata();
171//! assert_eq!(50, file_metadata.num_rows());
172//!
173//! let mut reader = ParquetRecordBatchReaderBuilder::try_new_with_options(file, options)
174//!   .unwrap()
175//!   .build()
176//!   .unwrap();
177//!
178//! let record_batch = reader.next().unwrap().unwrap();
179//! assert_eq!(50, record_batch.num_rows());
180//! ```

228/// A [`ProjectionMask`] identifies a set of columns within a potentially nested schema to project
229///
230/// In particular, a [`ProjectionMask`] can be constructed from a list of leaf column indices
231/// or root column indices where:
232///
233/// * Root columns are the direct children of the root schema, enumerated in order
234/// * Leaf columns are the child-less leaves of the schema as enumerated by a depth-first search
235///
236/// For example, the schema
237///
238/// ```ignore
239/// message schema {
240///   REQUIRED boolean         leaf_1;
241///   REQUIRED GROUP group {
242///     OPTIONAL int32 leaf_2;
243///     OPTIONAL int64 leaf_3;
244///   }
245/// }
246/// ```
247///
248/// Has roots `["leaf_1", "group"]` and leaves `["leaf_1", "leaf_2", "leaf_3"]`
249///
250/// For non-nested schemas, i.e. those containing only primitive columns, the root
251/// and leaves are the same
252///
253#[derive(Debug, Clone, PartialEq, Eq)]
254pub struct ProjectionMask {
255    /// If `Some`, a leaf column should be included if the value at
256    /// the corresponding index is true
257    ///
258    /// If `None`, all columns should be included
259    ///
260    /// # Examples
261    ///
262    /// Given the original parquet schema with leaf columns is `[a, b, c, d]`
263    ///
264    /// A mask of `[true, false, true, false]` will result in a schema 2
265    /// elements long:
266    /// * `fields[0]`: `a`
267    /// * `fields[1]`: `c`    
268    ///
269    /// A mask of `None` will result in a schema 4 elements long:
270    /// * `fields[0]`: `a`
271    /// * `fields[1]`: `b`
272    /// * `fields[2]`: `c`
273    /// * `fields[3]`: `d`
274    mask: Option<Vec<bool>>,
275}
276
277impl ProjectionMask {
278    /// Create a [`ProjectionMask`] which selects all columns
279    pub fn all() -> Self {
280        Self { mask: None }
281    }
282
283    /// Create a [`ProjectionMask`] which selects no columns
284    pub fn none(len: usize) -> Self {
285        Self {
286            mask: Some(vec![false; len]),
287        }
288    }
289
290    /// Create a [`ProjectionMask`] which selects only the specified leaf columns
291    ///
292    /// Note: repeated or out of order indices will not impact the final mask
293    ///
294    /// i.e. `[0, 1, 2]` will construct the same mask as `[1, 0, 0, 2]`
295    pub fn leaves(schema: &SchemaDescriptor, indices: impl IntoIterator<Item = usize>) -> Self {
296        let mut mask = vec![false; schema.num_columns()];
297        for leaf_idx in indices {
298            mask[leaf_idx] = true;
299        }
300        Self { mask: Some(mask) }
301    }
302
303    /// Create a [`ProjectionMask`] which selects only the specified root columns
304    ///
305    /// Note: repeated or out of order indices will not impact the final mask
306    ///
307    /// i.e. `[0, 1, 2]` will construct the same mask as `[1, 0, 0, 2]`
308    pub fn roots(schema: &SchemaDescriptor, indices: impl IntoIterator<Item = usize>) -> Self {
309        let num_root_columns = schema.root_schema().get_fields().len();
310        let mut root_mask = vec![false; num_root_columns];
311        for root_idx in indices {
312            root_mask[root_idx] = true;
313        }
314
315        let mask = (0..schema.num_columns())
316            .map(|leaf_idx| {
317                let root_idx = schema.get_column_root_idx(leaf_idx);
318                root_mask[root_idx]
319            })
320            .collect();
321
322        Self { mask: Some(mask) }
323    }
324
325    /// Create a [`ProjectionMask`] which selects only the named columns
326    ///
327    /// All leaf columns that fall below a given name will be selected. For example, given
328    /// the schema
329    /// ```ignore
330    /// message schema {
331    ///   OPTIONAL group a (MAP) {
332    ///     REPEATED group key_value {
333    ///       REQUIRED BYTE_ARRAY key (UTF8);  // leaf index 0
334    ///       OPTIONAL group value (MAP) {
335    ///         REPEATED group key_value {
336    ///           REQUIRED INT32 key;          // leaf index 1
337    ///           REQUIRED BOOLEAN value;      // leaf index 2
338    ///         }
339    ///       }
340    ///     }
341    ///   }
342    ///   REQUIRED INT32 b;                    // leaf index 3
343    ///   REQUIRED DOUBLE c;                   // leaf index 4
344    /// }
345    /// ```
346    /// `["a.key_value.value", "c"]` would return leaf columns 1, 2, and 4. `["a"]` would return
347    /// columns 0, 1, and 2.
348    ///
349    /// Note: repeated or out of order indices will not impact the final mask.
350    ///
351    /// i.e. `["b", "c"]` will construct the same mask as `["c", "b", "c"]`.
352    ///
353    /// Also, this will not produce the desired results if a column contains a '.' in its name.
354    /// Use [`Self::leaves`] or [`Self::roots`] in that case.
355    pub fn columns<'a>(
356        schema: &SchemaDescriptor,
357        names: impl IntoIterator<Item = &'a str>,
358    ) -> Self {
359        let mut mask = vec![false; schema.num_columns()];
360        for name in names {
361            let name_path: Vec<&str> = name.split('.').collect();
362            for (idx, col) in schema.columns().iter().enumerate() {
363                let path = col.path().parts();
364                // searching for "a.b.c" cannot match "a.b"
365                if name_path.len() > path.len() {
366                    continue;
367                }
368                // now path >= name_path, so check that each element in name_path matches
369                if name_path.iter().zip(path.iter()).all(|(a, b)| a == b) {
370                    mask[idx] = true;
371                }
372            }
373        }
374
375        Self { mask: Some(mask) }
376    }
377
378    /// Returns true if the leaf column `leaf_idx` is included by the mask
379    pub fn leaf_included(&self, leaf_idx: usize) -> bool {
380        self.mask.as_ref().map(|m| m[leaf_idx]).unwrap_or(true)
381    }
382
383    /// Union two projection masks
384    ///
385    /// Example:
386    /// ```text
387    /// mask1 = [true, false, true]
388    /// mask2 = [false, true, true]
389    /// union(mask1, mask2) = [true, true, true]
390    /// ```
391    pub fn union(&mut self, other: &Self) {
392        match (self.mask.as_ref(), other.mask.as_ref()) {
393            (None, _) | (_, None) => self.mask = None,
394            (Some(a), Some(b)) => {
395                debug_assert_eq!(a.len(), b.len());
396                let mask = a.iter().zip(b.iter()).map(|(&a, &b)| a || b).collect();
397                self.mask = Some(mask);
398            }
399        }
400    }
401
402    /// Intersect two projection masks
403    ///
404    /// Example:
405    /// ```text
406    /// mask1 = [true, false, true]
407    /// mask2 = [false, true, true]
408    /// intersect(mask1, mask2) = [false, false, true]
409    /// ```
410    pub fn intersect(&mut self, other: &Self) {
411        match (self.mask.as_ref(), other.mask.as_ref()) {
412            (None, _) => self.mask = other.mask.clone(),
413            (_, None) => {}
414            (Some(a), Some(b)) => {
415                debug_assert_eq!(a.len(), b.len());
416                let mask = a.iter().zip(b.iter()).map(|(&a, &b)| a && b).collect();
417                self.mask = Some(mask);
418            }
419        }
420    }
421}

423/// Lookups up the parquet column by name
424///
425/// Returns the parquet column index and the corresponding arrow field
426pub fn parquet_column<'a>(
427    parquet_schema: &SchemaDescriptor,
428    arrow_schema: &'a Schema,
429    name: &str,
430) -> Option<(usize, &'a FieldRef)> {
431    let (root_idx, field) = arrow_schema.fields.find(name)?;
432    if field.data_type().is_nested() {
433        // Nested fields are not supported and require non-trivial logic
434        // to correctly walk the parquet schema accounting for the
435        // logical type rules - <https://github.com/apache/parquet-format/blob/master/LogicalTypes.md>
436        //
437        // For example a ListArray could correspond to anything from 1 to 3 levels
438        // in the parquet schema
439        return None;
440    }
441
442    // This could be made more efficient (#TBD)
443    let parquet_idx = (0..parquet_schema.columns().len())
444        .find(|x| parquet_schema.get_column_root_idx(*x) == root_idx)?;
445    Some((parquet_idx, field))
446}

538    /// Sets the page index offset locations in the metadata to `None`
539    ///
540    /// This is because the offsets are used to find the relative location of the index
541    /// structures, and thus differ depending on how the structures are stored.
542    fn normalize_locations(metadata: ParquetMetaData) -> ParquetMetaData {
543        let mut metadata_builder = metadata.into_builder();
544        for rg in metadata_builder.take_row_groups() {
545            let mut rg_builder = rg.into_builder();
546            for col in rg_builder.take_columns() {
547                rg_builder = rg_builder.add_column_metadata(
548                    col.into_builder()
549                        .set_offset_index_offset(None)
550                        .set_index_page_offset(None)
551                        .set_column_index_offset(None)
552                        .build()
553                        .unwrap(),
554                );
555            }
556            let rg = rg_builder.build().unwrap();
557            metadata_builder = metadata_builder.add_row_group(rg);
558        }
559        metadata_builder.build()
560    }

562    /// Write a parquet filed into an in memory buffer
563    fn create_parquet_file() -> Bytes {
564        let mut buf = vec![];
565        let data = vec![100, 200, 201, 300, 102, 33];
566        let array: ArrayRef = Arc::new(Int32Array::from(data));
567        let batch = RecordBatch::try_from_iter(vec![("id", array)]).unwrap();
568        let props = WriterProperties::builder()
569            .set_statistics_enabled(EnabledStatistics::Page)
570            .set_write_page_header_statistics(true)
571            .build();
572
573        let mut writer = ArrowWriter::try_new(&mut buf, batch.schema(), Some(props)).unwrap();
574        writer.write(&batch).unwrap();
575        writer.finish().unwrap();
576        drop(writer);
577
578        Bytes::from(buf)
579    }
