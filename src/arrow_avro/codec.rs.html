31/// Avro types are not nullable, with nullability instead encoded as a union
32/// where one of the variants is the null type.
33///
34/// To accommodate this we special case two-variant unions where one of the
35/// variants is the null type, and use this to derive arrow's notion of nullability
36#[derive(Debug, Copy, Clone, PartialEq)]
37pub enum Nullability {
38    /// The nulls are encoded as the first union variant
39    NullFirst,
40    /// The nulls are encoded as the second union variant
41    NullSecond,
42}

44/// Contains information about how to resolve differences between a writer's and a reader's schema.

141/// An Avro datatype mapped to the arrow data model
142#[derive(Debug, Clone, PartialEq)]
143pub struct AvroDataType {
144    nullability: Option<Nullability>,
145    metadata: HashMap<String, String>,
146    codec: Codec,
147    pub(crate) resolution: Option<ResolutionInfo>,
148}
149
150impl AvroDataType {
151    /// Create a new [`AvroDataType`] with the given parts.
152    pub fn new(
153        codec: Codec,
154        metadata: HashMap<String, String>,
155        nullability: Option<Nullability>,
156    ) -> Self {
157        AvroDataType {
158            codec,
159            metadata,
160            nullability,
161            resolution: None,
162        }
163    }
164
165    #[inline]
166    fn new_with_resolution(
167        codec: Codec,
168        metadata: HashMap<String, String>,
169        nullability: Option<Nullability>,
170        resolution: Option<ResolutionInfo>,
171    ) -> Self {
172        Self {
173            codec,
174            metadata,
175            nullability,
176            resolution,
177        }
178    }
179
180    /// Returns an arrow [`Field`] with the given name
181    pub fn field_with_name(&self, name: &str) -> Field {
182        let nullable = self.nullability.is_some();
183        let data_type = self.codec.data_type();
184        let field = Field::new(name, data_type, nullable).with_metadata(self.metadata.clone());
185        #[cfg(feature = "canonical_extension_types")]
186        return with_extension_type(&self.codec, field);
187        #[cfg(not(feature = "canonical_extension_types"))]
188        field
189    }
190
191    /// Returns a reference to the codec used by this data type
192    ///
193    /// The codec determines how Avro data is encoded and mapped to Arrow data types.
194    /// This is useful when we need to inspect or use the specific encoding of a field.
195    pub fn codec(&self) -> &Codec {
196        &self.codec
197    }
198
199    /// Returns the nullability status of this data type
200    ///
201    /// In Avro, nullability is represented through unions with null types.
202    /// The returned value indicates how nulls are encoded in the Avro format:
203    /// - `Some(Nullability::NullFirst)` - Nulls are encoded as the first union variant
204    /// - `Some(Nullability::NullSecond)` - Nulls are encoded as the second union variant
205    /// - `None` - The type is not nullable
206    pub fn nullability(&self) -> Option<Nullability> {
207        self.nullability
208    }
209}

211/// A named [`AvroDataType`]
212#[derive(Debug, Clone, PartialEq)]
213pub struct AvroField {
214    name: String,
215    data_type: AvroDataType,
216}
217
218impl AvroField {
219    /// Returns the arrow [`Field`]
220    pub fn field(&self) -> Field {
221        self.data_type.field_with_name(&self.name)
222    }
223
224    /// Returns the [`AvroDataType`]
225    pub fn data_type(&self) -> &AvroDataType {
226        &self.data_type
227    }
228
229    /// Returns a new [`AvroField`] with Utf8View support enabled
230    ///
231    /// This will convert any Utf8 codecs to Utf8View codecs. This method is used to
232    /// enable potential performance optimizations in string-heavy workloads by using
233    /// Arrow's StringViewArray data structure.
234    ///
235    /// Returns a new `AvroField` with the same structure, but with string types
236    /// converted to use `Utf8View` instead of `Utf8`.
237    pub fn with_utf8view(&self) -> Self {
238        let mut field = self.clone();
239        if let Codec::Utf8 = field.data_type.codec {
240            field.data_type.codec = Codec::Utf8View;
241        }
242        field
243    }
244
245    /// Returns the name of this Avro field
246    ///
247    /// This is the field name as defined in the Avro schema.
248    /// It's used to identify fields within a record structure.
249    pub fn name(&self) -> &str {
250        &self.name
251    }
252
253    /// Performs schema resolution between a writer and reader schema.
254    ///
255    /// This is the primary entry point for handling schema evolution. It produces an
256    /// `AvroField` that contains all the necessary information to read data written
257    /// with the `writer` schema as if it were written with the `reader` schema.
258    pub(crate) fn resolve_from_writer_and_reader<'a>(
259        writer_schema: &'a Schema<'a>,
260        reader_schema: &'a Schema<'a>,
261        use_utf8view: bool,
262        strict_mode: bool,
263    ) -> Result<Self, ArrowError> {
264        let top_name = match reader_schema {
265            Schema::Complex(ComplexType::Record(r)) => r.name.to_string(),
266            _ => "root".to_string(),
267        };
268        let mut resolver = Maker::new(use_utf8view, strict_mode);
269        let data_type = resolver.make_data_type(writer_schema, Some(reader_schema), None)?;
270        Ok(Self {
271            name: top_name,
272            data_type,
273        })
274    }
275}

277impl<'a> TryFrom<&Schema<'a>> for AvroField {
278    type Error = ArrowError;
279
280    fn try_from(schema: &Schema<'a>) -> Result<Self, Self::Error> {
281        match schema {
282            Schema::Complex(ComplexType::Record(r)) => {
283                let mut resolver = Maker::new(false, false);
284                let data_type = resolver.make_data_type(schema, None, None)?;
285                Ok(AvroField {
286                    data_type,
287                    name: r.name.to_string(),
288                })
289            }
290            _ => Err(ArrowError::ParseError(format!(
291                "Expected record got {schema:?}"
292            ))),
293        }
294    }
295}

297/// Builder for an [`AvroField`]
298#[derive(Debug)]
299pub struct AvroFieldBuilder<'a> {
300    writer_schema: &'a Schema<'a>,
301    reader_schema: Option<&'a Schema<'a>>,
302    use_utf8view: bool,
303    strict_mode: bool,
304}
305
306impl<'a> AvroFieldBuilder<'a> {
307    /// Creates a new [`AvroFieldBuilder`] for a given writer schema.
308    pub fn new(writer_schema: &'a Schema<'a>) -> Self {
309        Self {
310            writer_schema,
311            reader_schema: None,
312            use_utf8view: false,
313            strict_mode: false,
314        }
315    }
316
317    /// Sets the reader schema for schema resolution.
318    ///
319    /// If a reader schema is provided, the builder will produce a resolved `AvroField`
320    /// that can handle differences between the writer's and reader's schemas.
321    #[inline]
322    pub fn with_reader_schema(mut self, reader_schema: &'a Schema<'a>) -> Self {
323        self.reader_schema = Some(reader_schema);
324        self
325    }
326
327    /// Enable or disable Utf8View support
328    pub fn with_utf8view(mut self, use_utf8view: bool) -> Self {
329        self.use_utf8view = use_utf8view;
330        self
331    }
332
333    /// Enable or disable strict mode.
334    pub fn with_strict_mode(mut self, strict_mode: bool) -> Self {
335        self.strict_mode = strict_mode;
336        self
337    }
338
339    /// Build an [`AvroField`] from the builder
340    pub fn build(self) -> Result<AvroField, ArrowError> {
341        match self.writer_schema {
342            Schema::Complex(ComplexType::Record(r)) => {
343                let mut resolver = Maker::new(self.use_utf8view, self.strict_mode);
344                let data_type =
345                    resolver.make_data_type(self.writer_schema, self.reader_schema, None)?;
346                Ok(AvroField {
347                    name: r.name.to_string(),
348                    data_type,
349                })
350            }
351            _ => Err(ArrowError::ParseError(format!(
352                "Expected a Record schema to build an AvroField, but got {:?}",
353                self.writer_schema
354            ))),
355        }
356    }
357}

359/// An Avro encoding
360///
361/// <https://avro.apache.org/docs/1.11.1/specification/#encodings>
362#[derive(Debug, Clone, PartialEq)]
363pub enum Codec {
364    /// Represents Avro null type, maps to Arrow's Null data type
365    Null,
366    /// Represents Avro boolean type, maps to Arrow's Boolean data type
367    Boolean,
368    /// Represents Avro int type, maps to Arrow's Int32 data type
369    Int32,
370    /// Represents Avro long type, maps to Arrow's Int64 data type
371    Int64,
372    /// Represents Avro float type, maps to Arrow's Float32 data type
373    Float32,
374    /// Represents Avro double type, maps to Arrow's Float64 data type
375    Float64,
376    /// Represents Avro bytes type, maps to Arrow's Binary data type
377    Binary,
378    /// String data represented as UTF-8 encoded bytes, corresponding to Arrow's StringArray
379    Utf8,
380    /// String data represented as UTF-8 encoded bytes with an optimized view representation,
381    /// corresponding to Arrow's StringViewArray which provides better performance for string operations
382    ///
383    /// The Utf8View option can be enabled via `ReadOptions::use_utf8view`.
384    Utf8View,
385    /// Represents Avro date logical type, maps to Arrow's Date32 data type
386    Date32,
387    /// Represents Avro time-millis logical type, maps to Arrow's Time32(TimeUnit::Millisecond) data type
388    TimeMillis,
389    /// Represents Avro time-micros logical type, maps to Arrow's Time64(TimeUnit::Microsecond) data type
390    TimeMicros,
391    /// Represents Avro timestamp-millis or local-timestamp-millis logical type
392    ///
393    /// Maps to Arrow's Timestamp(TimeUnit::Millisecond) data type
394    /// The boolean parameter indicates whether the timestamp has a UTC timezone (true) or is local time (false)
395    TimestampMillis(bool),
396    /// Represents Avro timestamp-micros or local-timestamp-micros logical type
397    ///
398    /// Maps to Arrow's Timestamp(TimeUnit::Microsecond) data type
399    /// The boolean parameter indicates whether the timestamp has a UTC timezone (true) or is local time (false)
400    TimestampMicros(bool),
401    /// Represents Avro fixed type, maps to Arrow's FixedSizeBinary data type
402    /// The i32 parameter indicates the fixed binary size
403    Fixed(i32),
404    /// Represents Avro decimal type, maps to Arrow's Decimal128 or Decimal256 data types
405    ///
406    /// The fields are `(precision, scale, fixed_size)`.
407    /// - `precision` (`usize`): Total number of digits.
408    /// - `scale` (`Option<usize>`): Number of fractional digits.
409    /// - `fixed_size` (`Option<usize>`): Size in bytes if backed by a `fixed` type, otherwise `None`.
410    Decimal(usize, Option<usize>, Option<usize>),
411    /// Represents Avro Uuid type, a FixedSizeBinary with a length of 16.
412    Uuid,
413    /// Represents an Avro enum, maps to Arrow's Dictionary(Int32, Utf8) type.
414    ///
415    /// The enclosed value contains the enum's symbols.
416    Enum(Arc<[String]>),
417    /// Represents Avro array type, maps to Arrow's List data type
418    List(Arc<AvroDataType>),
419    /// Represents Avro record type, maps to Arrow's Struct data type
420    Struct(Arc<[AvroField]>),
421    /// Represents Avro map type, maps to Arrow's Map data type
422    Map(Arc<AvroDataType>),
423    /// Represents Avro duration logical type, maps to Arrow's Interval(IntervalUnit::MonthDayNano) data type
424    Interval,
425}
426
427impl Codec {
428    fn data_type(&self) -> DataType {
429        match self {
430            Self::Null => DataType::Null,
431            Self::Boolean => DataType::Boolean,
432            Self::Int32 => DataType::Int32,
433            Self::Int64 => DataType::Int64,
434            Self::Float32 => DataType::Float32,
435            Self::Float64 => DataType::Float64,
436            Self::Binary => DataType::Binary,
437            Self::Utf8 => DataType::Utf8,
438            Self::Utf8View => DataType::Utf8View,
439            Self::Date32 => DataType::Date32,
440            Self::TimeMillis => DataType::Time32(TimeUnit::Millisecond),
441            Self::TimeMicros => DataType::Time64(TimeUnit::Microsecond),
442            Self::TimestampMillis(is_utc) => {
443                DataType::Timestamp(TimeUnit::Millisecond, is_utc.then(|| "+00:00".into()))
444            }
445            Self::TimestampMicros(is_utc) => {
446                DataType::Timestamp(TimeUnit::Microsecond, is_utc.then(|| "+00:00".into()))
447            }
448            Self::Interval => DataType::Interval(IntervalUnit::MonthDayNano),
449            Self::Fixed(size) => DataType::FixedSizeBinary(*size),
450            Self::Decimal(precision, scale, size) => {
451                let p = *precision as u8;
452                let s = scale.unwrap_or(0) as i8;
453                let too_large_for_128 = match *size {
454                    Some(sz) => sz > 16,
455                    None => {
456                        (p as usize) > DECIMAL128_MAX_PRECISION as usize
457                            || (s as usize) > DECIMAL128_MAX_SCALE as usize
458                    }
459                };
460                if too_large_for_128 {
461                    DataType::Decimal256(p, s)
462                } else {
463                    DataType::Decimal128(p, s)
464                }
465            }
466            Self::Uuid => DataType::FixedSizeBinary(16),
467            Self::Enum(_) => {
468                DataType::Dictionary(Box::new(DataType::Int32), Box::new(DataType::Utf8))
469            }
470            Self::List(f) => {
471                DataType::List(Arc::new(f.field_with_name(Field::LIST_FIELD_DEFAULT_NAME)))
472            }
473            Self::Struct(f) => DataType::Struct(f.iter().map(|x| x.field()).collect()),
474            Self::Map(value_type) => {
475                let val_dt = value_type.codec.data_type();
476                let val_field = Field::new("value", val_dt, value_type.nullability.is_some())
477                    .with_metadata(value_type.metadata.clone());
478                DataType::Map(
479                    Arc::new(Field::new(
480                        "entries",
481                        DataType::Struct(Fields::from(vec![
482                            Field::new("key", DataType::Utf8, false),
483                            val_field,
484                        ])),
485                        false,
486                    )),
487                    false,
488                )
489            }
490        }
491    }
492}

509fn parse_decimal_attributes(
510    attributes: &Attributes,
511    fallback_size: Option<usize>,
512    precision_required: bool,
513) -> Result<(usize, usize, Option<usize>), ArrowError> {
514    let precision = attributes
515        .additional
516        .get("precision")
517        .and_then(|v| v.as_u64())
518        .or(if precision_required { None } else { Some(10) })
519        .ok_or_else(|| ArrowError::ParseError("Decimal requires precision".to_string()))?
520        as usize;
521    let scale = attributes
522        .additional
523        .get("scale")
524        .and_then(|v| v.as_u64())
525        .unwrap_or(0) as usize;
526    let size = attributes
527        .additional
528        .get("size")
529        .and_then(|v| v.as_u64())
530        .map(|s| s as usize)
531        .or(fallback_size);
532    Ok((precision, scale, size))
533}

535impl Codec {
536    /// Converts a string codec to use Utf8View if requested
537    ///
538    /// The conversion only happens if both:
539    /// 1. `use_utf8view` is true
540    /// 2. The codec is currently `Utf8`
541    ///
542    /// # Example
543    /// ```
544    /// # use arrow_avro::codec::Codec;
545    /// let utf8_codec1 = Codec::Utf8;
546    /// let utf8_codec2 = Codec::Utf8;
547    ///
548    /// // Convert to Utf8View
549    /// let view_codec = utf8_codec1.with_utf8view(true);
550    /// assert!(matches!(view_codec, Codec::Utf8View));
551    ///
552    /// // Don't convert if use_utf8view is false
553    /// let unchanged_codec = utf8_codec2.with_utf8view(false);
554    /// assert!(matches!(unchanged_codec, Codec::Utf8));
555    /// ```
556    pub fn with_utf8view(self, use_utf8view: bool) -> Self {
557        if use_utf8view && matches!(self, Self::Utf8) {
558            Self::Utf8View
559        } else {
560            self
561        }
562    }
563}

565/// Resolves Avro type names to [`AvroDataType`]
566///
567/// See <https://avro.apache.org/docs/1.11.1/specification/#names>
568#[derive(Debug, Default)]
569struct Resolver<'a> {
570    map: HashMap<(&'a str, &'a str), AvroDataType>,
571}
572
573impl<'a> Resolver<'a> {
574    fn register(&mut self, name: &'a str, namespace: Option<&'a str>, schema: AvroDataType) {
575        self.map.insert((namespace.unwrap_or(""), name), schema);
576    }
577
578    fn resolve(&self, name: &str, namespace: Option<&'a str>) -> Result<AvroDataType, ArrowError> {
579        let (namespace, name) = name
580            .rsplit_once('.')
581            .unwrap_or_else(|| (namespace.unwrap_or(""), name));
582
583        self.map
584            .get(&(namespace, name))
585            .ok_or_else(|| ArrowError::ParseError(format!("Failed to resolve {namespace}.{name}")))
586            .cloned()
587    }
588}

590/// Resolves Avro type names to [`AvroDataType`]
591///
592/// See <https://avro.apache.org/docs/1.11.1/specification/#names>
593struct Maker<'a> {
594    resolver: Resolver<'a>,
595    use_utf8view: bool,
596    strict_mode: bool,
597}
598
599impl<'a> Maker<'a> {
600    fn new(use_utf8view: bool, strict_mode: bool) -> Self {
601        Self {
602            resolver: Default::default(),
603            use_utf8view,
604            strict_mode,
605        }
606    }
607    fn make_data_type<'s>(
608        &mut self,
609        writer_schema: &'s Schema<'a>,
610        reader_schema: Option<&'s Schema<'a>>,
611        namespace: Option<&'a str>,
612    ) -> Result<AvroDataType, ArrowError> {
613        match reader_schema {
614            Some(reader_schema) => self.resolve_type(writer_schema, reader_schema, namespace),
615            None => self.parse_type(writer_schema, namespace),
616        }
617    }
618
619    /// Parses a [`AvroDataType`] from the provided [`Schema`] and the given `name` and `namespace`
620    ///
621    /// `name`: is the name used to refer to `schema` in its parent
622    /// `namespace`: an optional qualifier used as part of a type hierarchy
623    /// If the data type is a string, convert to use Utf8View if requested
624    ///
625    /// This function is used during the schema conversion process to determine whether
626    /// string data should be represented as StringArray (default) or StringViewArray.
627    ///
628    /// `use_utf8view`: if true, use Utf8View instead of Utf8 for string types
629    ///
630    /// See [`Resolver`] for more information
631    fn parse_type<'s>(
632        &mut self,
633        schema: &'s Schema<'a>,
634        namespace: Option<&'a str>,
635    ) -> Result<AvroDataType, ArrowError> {
636        match schema {
637            Schema::TypeName(TypeName::Primitive(p)) => Ok(AvroDataType::new(
638                Codec::from(*p).with_utf8view(self.use_utf8view),
639                Default::default(),
640                None,
641            )),
642            Schema::TypeName(TypeName::Ref(name)) => self.resolver.resolve(name, namespace),
643            Schema::Union(f) => {
644                // Special case the common case of nullable primitives
645                let null = f
646                    .iter()
647                    .position(|x| x == &Schema::TypeName(TypeName::Primitive(PrimitiveType::Null)));
648                match (f.len() == 2, null) {
649                    (true, Some(0)) => {
650                        let mut field = self.parse_type(&f[1], namespace)?;
651                        field.nullability = Some(Nullability::NullFirst);
652                        Ok(field)
653                    }
654                    (true, Some(1)) => {
655                        if self.strict_mode {
656                            return Err(ArrowError::SchemaError(
657                                "Found Avro union of the form ['T','null'], which is disallowed in strict_mode"
658                                    .to_string(),
659                            ));
660                        }
661                        let mut field = self.parse_type(&f[0], namespace)?;
662                        field.nullability = Some(Nullability::NullSecond);
663                        Ok(field)
664                    }
665                    _ => Err(ArrowError::NotYetImplemented(format!(
666                        "Union of {f:?} not currently supported"
667                    ))),
668                }
669            }
670            Schema::Complex(c) => match c {
671                ComplexType::Record(r) => {
672                    let namespace = r.namespace.or(namespace);
673                    let fields = r
674                        .fields
675                        .iter()
676                        .map(|field| {
677                            Ok(AvroField {
678                                name: field.name.to_string(),
679                                data_type: self.parse_type(&field.r#type, namespace)?,
680                            })
681                        })
682                        .collect::<Result<_, ArrowError>>()?;
683                    let field = AvroDataType {
684                        nullability: None,
685                        codec: Codec::Struct(fields),
686                        metadata: r.attributes.field_metadata(),
687                        resolution: None,
688                    };
689                    self.resolver.register(r.name, namespace, field.clone());
690                    Ok(field)
691                }
692                ComplexType::Array(a) => {
693                    let mut field = self.parse_type(a.items.as_ref(), namespace)?;
694                    Ok(AvroDataType {
695                        nullability: None,
696                        metadata: a.attributes.field_metadata(),
697                        codec: Codec::List(Arc::new(field)),
698                        resolution: None,
699                    })
700                }
701                ComplexType::Fixed(f) => {
702                    let size = f.size.try_into().map_err(|e| {
703                        ArrowError::ParseError(format!("Overflow converting size to i32: {e}"))
704                    })?;
705                    let md = f.attributes.field_metadata();
706                    let field = match f.attributes.logical_type {
707                        Some("decimal") => {
708                            let (precision, scale, _) =
709                                parse_decimal_attributes(&f.attributes, Some(size as usize), true)?;
710                            AvroDataType {
711                                nullability: None,
712                                metadata: md,
713                                codec: Codec::Decimal(precision, Some(scale), Some(size as usize)),
714                                resolution: None,
715                            }
716                        }
717                        Some("duration") => {
718                            if size != 12 {
719                                return Err(ArrowError::ParseError(format!(
720                                    "Invalid fixed size for Duration: {size}, must be 12"
721                                )));
722                            };
723                            AvroDataType {
724                                nullability: None,
725                                metadata: md,
726                                codec: Codec::Interval,
727                                resolution: None,
728                            }
729                        }
730                        _ => AvroDataType {
731                            nullability: None,
732                            metadata: md,
733                            codec: Codec::Fixed(size),
734                            resolution: None,
735                        },
736                    };
737                    self.resolver.register(f.name, namespace, field.clone());
738                    Ok(field)
739                }
740                ComplexType::Enum(e) => {
741                    let namespace = e.namespace.or(namespace);
742                    let symbols = e
743                        .symbols
744                        .iter()
745                        .map(|s| s.to_string())
746                        .collect::<Arc<[String]>>();
747
748                    let mut metadata = e.attributes.field_metadata();
749                    let symbols_json = serde_json::to_string(&e.symbols).map_err(|e| {
750                        ArrowError::ParseError(format!("Failed to serialize enum symbols: {e}"))
751                    })?;
752                    metadata.insert(AVRO_ENUM_SYMBOLS_METADATA_KEY.to_string(), symbols_json);
753                    let field = AvroDataType {
754                        nullability: None,
755                        metadata,
756                        codec: Codec::Enum(symbols),
757                        resolution: None,
758                    };
759                    self.resolver.register(e.name, namespace, field.clone());
760                    Ok(field)
761                }
762                ComplexType::Map(m) => {
763                    let val = self.parse_type(&m.values, namespace)?;
764                    Ok(AvroDataType {
765                        nullability: None,
766                        metadata: m.attributes.field_metadata(),
767                        codec: Codec::Map(Arc::new(val)),
768                        resolution: None,
769                    })
770                }
771            },
772            Schema::Type(t) => {
773                let mut field = self.parse_type(&Schema::TypeName(t.r#type.clone()), namespace)?;
774                // https://avro.apache.org/docs/1.11.1/specification/#logical-types
775                match (t.attributes.logical_type, &mut field.codec) {
776                    (Some("decimal"), c @ Codec::Binary) => {
777                        let (prec, sc, _) = parse_decimal_attributes(&t.attributes, None, false)?;
778                        *c = Codec::Decimal(prec, Some(sc), None);
779                    }
780                    (Some("date"), c @ Codec::Int32) => *c = Codec::Date32,
781                    (Some("time-millis"), c @ Codec::Int32) => *c = Codec::TimeMillis,
782                    (Some("time-micros"), c @ Codec::Int64) => *c = Codec::TimeMicros,
783                    (Some("timestamp-millis"), c @ Codec::Int64) => {
784                        *c = Codec::TimestampMillis(true)
785                    }
786                    (Some("timestamp-micros"), c @ Codec::Int64) => {
787                        *c = Codec::TimestampMicros(true)
788                    }
789                    (Some("local-timestamp-millis"), c @ Codec::Int64) => {
790                        *c = Codec::TimestampMillis(false)
791                    }
792                    (Some("local-timestamp-micros"), c @ Codec::Int64) => {
793                        *c = Codec::TimestampMicros(false)
794                    }
795                    (Some("uuid"), c @ Codec::Utf8) => *c = Codec::Uuid,
796                    (Some(logical), _) => {
797                        // Insert unrecognized logical type into metadata map
798                        field.metadata.insert("logicalType".into(), logical.into());
799                    }
800                    (None, _) => {}
801                }
802                if !t.attributes.additional.is_empty() {
803                    for (k, v) in &t.attributes.additional {
804                        field.metadata.insert(k.to_string(), v.to_string());
805                    }
806                }
807                Ok(field)
808            }
809        }
810    }

812    fn resolve_type<'s>(
813        &mut self,
814        writer_schema: &'s Schema<'a>,
815        reader_schema: &'s Schema<'a>,
816        namespace: Option<&'a str>,
817    ) -> Result<AvroDataType, ArrowError> {
818        match (writer_schema, reader_schema) {
819            (
820                Schema::TypeName(TypeName::Primitive(writer_primitive)),
821                Schema::TypeName(TypeName::Primitive(reader_primitive)),
822            ) => self.resolve_primitives(*writer_primitive, *reader_primitive, reader_schema),
823            (
824                Schema::Type(Type {
825                    r#type: TypeName::Primitive(writer_primitive),
826                    ..
827                }),
828                Schema::Type(Type {
829                    r#type: TypeName::Primitive(reader_primitive),
830                    ..
831                }),
832            ) => self.resolve_primitives(*writer_primitive, *reader_primitive, reader_schema),
833            (
834                Schema::TypeName(TypeName::Primitive(writer_primitive)),
835                Schema::Type(Type {
836                    r#type: TypeName::Primitive(reader_primitive),
837                    ..
838                }),
839            ) => self.resolve_primitives(*writer_primitive, *reader_primitive, reader_schema),
840            (
841                Schema::Type(Type {
842                    r#type: TypeName::Primitive(writer_primitive),
843                    ..
844                }),
845                Schema::TypeName(TypeName::Primitive(reader_primitive)),
846            ) => self.resolve_primitives(*writer_primitive, *reader_primitive, reader_schema),
847            (
848                Schema::Complex(ComplexType::Record(writer_record)),
849                Schema::Complex(ComplexType::Record(reader_record)),
850            ) => self.resolve_records(writer_record, reader_record, namespace),
851            (Schema::Union(writer_variants), Schema::Union(reader_variants)) => {
852                self.resolve_nullable_union(writer_variants, reader_variants, namespace)
853            }
854            _ => Err(ArrowError::NotYetImplemented(
855                "Other resolutions not yet implemented".to_string(),
856            )),
857        }
858    }
859
860    fn resolve_primitives(
861        &mut self,
862        write_primitive: PrimitiveType,
863        read_primitive: PrimitiveType,
864        reader_schema: &Schema<'a>,
865    ) -> Result<AvroDataType, ArrowError> {
866        if write_primitive == read_primitive {
867            return self.parse_type(reader_schema, None);
868        }
869        let promotion = match (write_primitive, read_primitive) {
870            (PrimitiveType::Int, PrimitiveType::Long) => Promotion::IntToLong,
871            (PrimitiveType::Int, PrimitiveType::Float) => Promotion::IntToFloat,
872            (PrimitiveType::Int, PrimitiveType::Double) => Promotion::IntToDouble,
873            (PrimitiveType::Long, PrimitiveType::Float) => Promotion::LongToFloat,
874            (PrimitiveType::Long, PrimitiveType::Double) => Promotion::LongToDouble,
875            (PrimitiveType::Float, PrimitiveType::Double) => Promotion::FloatToDouble,
876            (PrimitiveType::String, PrimitiveType::Bytes) => Promotion::StringToBytes,
877            (PrimitiveType::Bytes, PrimitiveType::String) => Promotion::BytesToString,
878            _ => {
879                return Err(ArrowError::ParseError(format!(
880                    "Illegal promotion {write_primitive:?} to {read_primitive:?}"
881                )))
882            }
883        };
884        let mut datatype = self.parse_type(reader_schema, None)?;
885        datatype.resolution = Some(ResolutionInfo::Promotion(promotion));
886        Ok(datatype)
887    }
888
889    fn resolve_nullable_union(
890        &mut self,
891        writer_variants: &[Schema<'a>],
892        reader_variants: &[Schema<'a>],
893        namespace: Option<&'a str>,
894    ) -> Result<AvroDataType, ArrowError> {
895        // Only support unions with exactly two branches, one of which is `null` on both sides
896        if writer_variants.len() != 2 || reader_variants.len() != 2 {
897            return Err(ArrowError::NotYetImplemented(
898                "Only 2-branch unions are supported for schema resolution".to_string(),
899            ));
900        }
901        let is_null = |s: &Schema<'a>| {
902            matches!(
903                s,
904                Schema::TypeName(TypeName::Primitive(PrimitiveType::Null))
905            )
906        };
907        let w_null_pos = writer_variants.iter().position(is_null);
908        let r_null_pos = reader_variants.iter().position(is_null);
909        match (w_null_pos, r_null_pos) {
910            (Some(wp), Some(rp)) => {
911                // Extract a non-null branch on each side
912                let w_nonnull = &writer_variants[1 - wp];
913                let r_nonnull = &reader_variants[1 - rp];
914                // Resolve the non-null branch
915                let mut dt = self.make_data_type(w_nonnull, Some(r_nonnull), namespace)?;
916                // Adopt reader union null ordering
917                dt.nullability = Some(match rp {
918                    0 => Nullability::NullFirst,
919                    1 => Nullability::NullSecond,
920                    _ => unreachable!(),
921                });
922                Ok(dt)
923            }
924            _ => Err(ArrowError::NotYetImplemented(
925                "Union resolution requires both writer and reader to be nullable unions"
926                    .to_string(),
927            )),
928        }
929    }
930
931    fn resolve_records(
932        &mut self,
933        writer_record: &Record<'a>,
934        reader_record: &Record<'a>,
935        namespace: Option<&'a str>,
936    ) -> Result<AvroDataType, ArrowError> {
937        // Names must match or be aliased
938        let names_match = writer_record.name == reader_record.name
939            || reader_record.aliases.contains(&writer_record.name)
940            || writer_record.aliases.contains(&reader_record.name);
941        if !names_match {
942            return Err(ArrowError::ParseError(format!(
943                "Record name mismatch writer={}, reader={}",
944                writer_record.name, reader_record.name
945            )));
946        }
947        let writer_ns = writer_record.namespace.or(namespace);
948        let reader_ns = reader_record.namespace.or(namespace);
949        // Map writer field name -> index
950        let mut writer_index_map =
951            HashMap::<&str, usize>::with_capacity(writer_record.fields.len());
952        for (idx, write_field) in writer_record.fields.iter().enumerate() {
953            writer_index_map.insert(write_field.name, idx);
954        }
955        // Prepare outputs
956        let mut reader_fields: Vec<AvroField> = Vec::with_capacity(reader_record.fields.len());
957        let mut writer_to_reader: Vec<Option<usize>> = vec![None; writer_record.fields.len()];
958        //let mut skip_fields: Vec<Option<AvroDataType>> = vec![None; writer_record.fields.len()];
959        //let mut default_fields: Vec<usize> = Vec::new();
960        // Build reader fields and mapping
961        for (reader_idx, r_field) in reader_record.fields.iter().enumerate() {
962            if let Some(&writer_idx) = writer_index_map.get(r_field.name) {
963                // Field exists in writer: resolve types (including promotions and union-of-null)
964                let w_schema = &writer_record.fields[writer_idx].r#type;
965                let resolved_dt =
966                    self.make_data_type(w_schema, Some(&r_field.r#type), reader_ns)?;
967                reader_fields.push(AvroField {
968                    name: r_field.name.to_string(),
969                    data_type: resolved_dt,
970                });
971                writer_to_reader[writer_idx] = Some(reader_idx);
972            } else {
973                return Err(ArrowError::NotYetImplemented(
974                    "New fields from reader with default values not yet implemented".to_string(),
975                ));
976            }
977        }
978        // Implement writer-only fields to skip in Follow-up PR here
979        // Build resolved record AvroDataType
980        let resolved = AvroDataType::new_with_resolution(
981            Codec::Struct(Arc::from(reader_fields)),
982            reader_record.attributes.field_metadata(),
983            None,
984            Some(ResolutionInfo::Record(ResolvedRecord {
985                writer_to_reader: Arc::from(writer_to_reader),
986                default_fields: Arc::default(),
987                skip_fields: Arc::default(),
988            })),
989        );
990        // Register a resolved record by reader name+namespace for potential named type refs
991        self.resolver
992            .register(reader_record.name, reader_ns, resolved.clone());
993        Ok(resolved)
994    }
