18//! Avro reader
19//!
20//! This module provides facilities to read Apache Avro-encoded files or streams
21//! into Arrow's `RecordBatch` format. In particular, it introduces:
22//!
23//! * `ReaderBuilder`: Configures Avro reading, e.g., batch size
24//! * `Reader`: Yields `RecordBatch` values, implementing `Iterator`
25//! * `Decoder`: A low-level push-based decoder for Avro records
26//!
27//! # Basic Usage
28//!
29//! `Reader` can be used directly with synchronous data sources, such as [`std::fs::File`].
30//!
31//! ## Reading a Single Batch
32//!
33//! ```
34//! # use std::fs::File;
35//! # use std::io::BufReader;
36//! # use arrow_avro::reader::ReaderBuilder;
37//! # let path = "avro/alltypes_plain.avro";
38//! # let path = match std::env::var("ARROW_TEST_DATA") {
39//! #   Ok(dir) => format!("{dir}/{path}"),
40//! #   Err(_) => format!("../testing/data/{path}")
41//! # };
42//! let file = File::open(path).unwrap();
43//! let mut avro = ReaderBuilder::new().build(BufReader::new(file)).unwrap();
44//! let batch = avro.next().unwrap();
45//! ```
46//!
47//! # Async Usage
48//!
49//! The lower-level `Decoder` can be integrated with various forms of async data streams,
50//! and is designed to be agnostic to different async IO primitives within
51//! the Rust ecosystem. It works by incrementally decoding Avro data from byte slices.
52//!
53//! For example, see below for how it could be used with an arbitrary `Stream` of `Bytes`:
54//!
55//! ```
56//! # use std::task::{Poll, ready};
57//! # use bytes::{Buf, Bytes};
58//! # use arrow_schema::ArrowError;
59//! # use futures::stream::{Stream, StreamExt};
60//! # use arrow_array::RecordBatch;
61//! # use arrow_avro::reader::Decoder;
62//!
63//! fn decode_stream<S: Stream<Item = Bytes> + Unpin>(
64//!     mut decoder: Decoder,
65//!     mut input: S,
66//! ) -> impl Stream<Item = Result<RecordBatch, ArrowError>> {
67//!     let mut buffered = Bytes::new();
68//!     futures::stream::poll_fn(move |cx| {
69//!         loop {
70//!             if buffered.is_empty() {
71//!                 buffered = match ready!(input.poll_next_unpin(cx)) {
72//!                     Some(b) => b,
73//!                     None => break,
74//!                 };
75//!             }
76//!             let decoded = match decoder.decode(buffered.as_ref()) {
77//!                 Ok(decoded) => decoded,
78//!                 Err(e) => return Poll::Ready(Some(Err(e))),
79//!             };
80//!             let read = buffered.len();
81//!             buffered.advance(decoded);
82//!             if decoded != read {
83//!                 break
84//!             }
85//!         }
86//!         // Convert any fully-decoded rows to a RecordBatch, if available
87//!         Poll::Ready(decoder.flush().transpose())
88//!     })
89//! }
90//! ```
91//!

112/// Read the Avro file header (magic, metadata, sync marker) from `reader`.
113fn read_header<R: BufRead>(mut reader: R) -> Result<Header, ArrowError> {
114    let mut decoder = HeaderDecoder::default();
115    loop {
116        let buf = reader.fill_buf()?;
117        if buf.is_empty() {
118            break;
119        }
120        let read = buf.len();
121        let decoded = decoder.decode(buf)?;
122        reader.consume(decoded);
123        if decoded != read {
124            break;
125        }
126    }
127    decoder.flush().ok_or_else(|| {
128        ArrowError::ParseError("Unexpected EOF while reading Avro header".to_string())
129    })
130}

141/// A low-level interface for decoding Avro-encoded bytes into Arrow `RecordBatch`.
142#[derive(Debug)]
143pub struct Decoder {
144    active_decoder: RecordDecoder,
145    active_fingerprint: Option<Fingerprint>,
146    batch_size: usize,
147    remaining_capacity: usize,
148    cache: IndexMap<Fingerprint, RecordDecoder>,
149    fingerprint_algorithm: FingerprintAlgorithm,
150    utf8_view: bool,
151    strict_mode: bool,
152    pending_schema: Option<(Fingerprint, RecordDecoder)>,
153    awaiting_body: bool,
154}
155
156impl Decoder {
157    /// Return the Arrow schema for the rows decoded by this decoder
158    pub fn schema(&self) -> SchemaRef {
159        self.active_decoder.schema().clone()
160    }
161
162    /// Return the configured maximum number of rows per batch
163    pub fn batch_size(&self) -> usize {
164        self.batch_size
165    }
166
167    /// Feed `data` into the decoder row by row until we either:
168    /// - consume all bytes in `data`, or
169    /// - reach `batch_size` decoded rows.
170    ///
171    /// Returns the number of bytes consumed.
172    pub fn decode(&mut self, data: &[u8]) -> Result<usize, ArrowError> {
173        let mut total_consumed = 0usize;
174        while total_consumed < data.len() && self.remaining_capacity > 0 {
175            if self.awaiting_body {
176                match self.active_decoder.decode(&data[total_consumed..], 1) {
177                    Ok(n) => {
178                        self.remaining_capacity -= 1;
179                        total_consumed += n;
180                        self.awaiting_body = false;
181                        continue;
182                    }
183                    Err(ref e) if is_incomplete_data(e) => break,
184                    err => return err,
185                };
186            }
187            match self.handle_prefix(&data[total_consumed..])? {
188                Some(0) => break, // insufficient bytes
189                Some(n) => {
190                    total_consumed += n;
191                    self.apply_pending_schema_if_batch_empty();
192                    self.awaiting_body = true;
193                }
194                None => {
195                    return Err(ArrowError::ParseError(
196                        "Missing magic bytes and fingerprint".to_string(),
197                    ))
198                }
199            }
200        }
201        Ok(total_consumed)
202    }
203
204    // Attempt to handle a single‑object‑encoding prefix at the current position.
205    //
206    // * Ok(None) – buffer does not start with the prefix.
207    // * Ok(Some(0)) – prefix detected, but the buffer is too short; caller should await more bytes.
208    // * Ok(Some(n)) – consumed `n > 0` bytes of a complete prefix (magic and fingerprint).
209    fn handle_prefix(&mut self, buf: &[u8]) -> Result<Option<usize>, ArrowError> {
210        // Need at least the magic bytes to decide (2 bytes).
211        let Some(magic_bytes) = buf.get(..SINGLE_OBJECT_MAGIC.len()) else {
212            return Ok(Some(0)); // Get more bytes
213        };
214        // Bail out early if the magic does not match.
215        if magic_bytes != SINGLE_OBJECT_MAGIC {
216            return Ok(None); // Continue to decode the next record
217        }
218        // Try to parse the fingerprint that follows the magic.
219        let fingerprint_size = match self.fingerprint_algorithm {
220            FingerprintAlgorithm::Rabin => self
221                .handle_fingerprint(&buf[SINGLE_OBJECT_MAGIC.len()..], |bytes| {
222                    Fingerprint::Rabin(u64::from_le_bytes(bytes))
223                })?,
224        };
225        // Convert the inner result into a “bytes consumed” count.
226        // NOTE: Incomplete fingerprint consumes no bytes.
227        let consumed = fingerprint_size.map_or(0, |n| n + SINGLE_OBJECT_MAGIC.len());
228        Ok(Some(consumed))
229    }
230
231    // Attempts to read and install a new fingerprint of `N` bytes.
232    //
233    // * Ok(None) – insufficient bytes (`buf.len() < `N`).
234    // * Ok(Some(N)) – fingerprint consumed (always `N`).
235    fn handle_fingerprint<const N: usize>(
236        &mut self,
237        buf: &[u8],
238        fingerprint_from: impl FnOnce([u8; N]) -> Fingerprint,
239    ) -> Result<Option<usize>, ArrowError> {
240        // Need enough bytes to get fingerprint (next N bytes)
241        let Some(fingerprint_bytes) = buf.get(..N) else {
242            return Ok(None); // Insufficient bytes
243        };
244        // SAFETY: length checked above.
245        let new_fingerprint = fingerprint_from(fingerprint_bytes.try_into().unwrap());
246        // If the fingerprint indicates a schema change, prepare to switch decoders.
247        if self.active_fingerprint != Some(new_fingerprint) {
248            let Some(new_decoder) = self.cache.shift_remove(&new_fingerprint) else {
249                return Err(ArrowError::ParseError(format!(
250                    "Unknown fingerprint: {new_fingerprint:?}"
251                )));
252            };
253            self.pending_schema = Some((new_fingerprint, new_decoder));
254            // If there are already decoded rows, we must flush them first.
255            // Reducing `remaining_capacity` to 0 ensures `flush` is called next.
256            if self.remaining_capacity < self.batch_size {
257                self.remaining_capacity = 0;
258            }
259        }
260        Ok(Some(N))
261    }
262
263    fn apply_pending_schema(&mut self) {
264        if let Some((new_fingerprint, new_decoder)) = self.pending_schema.take() {
265            if let Some(old_fingerprint) = self.active_fingerprint.replace(new_fingerprint) {
266                let old_decoder = std::mem::replace(&mut self.active_decoder, new_decoder);
267                self.cache.shift_remove(&old_fingerprint);
268                self.cache.insert(old_fingerprint, old_decoder);
269            } else {
270                self.active_decoder = new_decoder;
271            }
272        }
273    }
274
275    fn apply_pending_schema_if_batch_empty(&mut self) {
276        if self.batch_is_empty() {
277            self.apply_pending_schema();
278        }
279    }
280
281    fn flush_and_reset(&mut self) -> Result<Option<RecordBatch>, ArrowError> {
282        if self.batch_is_empty() {
283            return Ok(None);
284        }
285        let batch = self.active_decoder.flush()?;
286        self.remaining_capacity = self.batch_size;
287        Ok(Some(batch))
288    }
289
290    /// Produce a `RecordBatch` if at least one row is fully decoded, returning
291    /// `Ok(None)` if no new rows are available.
292    pub fn flush(&mut self) -> Result<Option<RecordBatch>, ArrowError> {
293        // We must flush the active decoder before switching to the pending one.
294        let batch = self.flush_and_reset();
295        self.apply_pending_schema();
296        batch
297    }
298
299    /// Returns the number of rows that can be added to this decoder before it is full.
300    pub fn capacity(&self) -> usize {
301        self.remaining_capacity
302    }
303
304    /// Returns true if the decoder has reached its capacity for the current batch.
305    pub fn batch_is_full(&self) -> bool {
306        self.remaining_capacity == 0
307    }
308
309    /// Returns true if the decoder has not decoded any batches yet.
310    pub fn batch_is_empty(&self) -> bool {
311        self.remaining_capacity == self.batch_size
312    }
313
314    // Decode either the block count or remaining capacity from `data` (an OCF block payload).
315    //
316    // Returns the number of bytes consumed from `data` along with the number of records decoded.
317    fn decode_block(&mut self, data: &[u8], count: usize) -> Result<(usize, usize), ArrowError> {
318        // OCF decoding never interleaves records across blocks, so no chunking.
319        let to_decode = std::cmp::min(count, self.remaining_capacity);
320        if to_decode == 0 {
321            return Ok((0, 0));
322        }
323        let consumed = self.active_decoder.decode(data, to_decode)?;
324        self.remaining_capacity -= to_decode;
325        Ok((consumed, to_decode))
326    }
327
328    // Produce a `RecordBatch` if at least one row is fully decoded, returning
329    // `Ok(None)` if no new rows are available.
330    fn flush_block(&mut self) -> Result<Option<RecordBatch>, ArrowError> {
331        self.flush_and_reset()
332    }
333}

360impl ReaderBuilder {

362    /// - `batch_size` = 1024
363    /// - `strict_mode` = false
364    /// - `utf8_view` = false
365    /// - `reader_schema` = None

372    fn make_record_decoder(
373        &self,
374        writer_schema: &Schema,
375        reader_schema: Option<&Schema>,
376    ) -> Result<RecordDecoder, ArrowError> {
377        let mut builder = AvroFieldBuilder::new(writer_schema);
378        if let Some(reader_schema) = reader_schema {
379            builder = builder.with_reader_schema(reader_schema);
380        }
381        let root = builder
382            .with_utf8view(self.utf8_view)
383            .with_strict_mode(self.strict_mode)
384            .build()?;
385        RecordDecoder::try_new_with_options(root.data_type(), self.utf8_view)
386    }
387
388    fn make_record_decoder_from_schemas(
389        &self,
390        writer_schema: &Schema,
391        reader_schema: Option<&AvroSchema>,
392    ) -> Result<RecordDecoder, ArrowError> {
393        let reader_schema_raw = reader_schema.map(|s| s.schema()).transpose()?;
394        self.make_record_decoder(writer_schema, reader_schema_raw.as_ref())
395    }
396
397    fn make_decoder_with_parts(
398        &self,
399        active_decoder: RecordDecoder,
400        active_fingerprint: Option<Fingerprint>,
401        cache: IndexMap<Fingerprint, RecordDecoder>,
402        fingerprint_algorithm: FingerprintAlgorithm,
403    ) -> Decoder {
404        Decoder {
405            batch_size: self.batch_size,
406            remaining_capacity: self.batch_size,
407            active_fingerprint,
408            active_decoder,
409            cache,
410            utf8_view: self.utf8_view,
411            fingerprint_algorithm,
412            strict_mode: self.strict_mode,
413            pending_schema: None,
414            awaiting_body: false,
415        }
416    }
417
418    fn make_decoder(
419        &self,
420        header: Option<&Header>,
421        reader_schema: Option<&AvroSchema>,
422    ) -> Result<Decoder, ArrowError> {
423        if let Some(hdr) = header {
424            let writer_schema = hdr
425                .schema()
426                .map_err(|e| ArrowError::ExternalError(Box::new(e)))?
427                .ok_or_else(|| {
428                    ArrowError::ParseError("No Avro schema present in file header".into())
429                })?;
430            let record_decoder =
431                self.make_record_decoder_from_schemas(&writer_schema, reader_schema)?;
432            return Ok(self.make_decoder_with_parts(
433                record_decoder,
434                None,
435                IndexMap::new(),
436                FingerprintAlgorithm::Rabin,
437            ));
438        }
439        let store = self.writer_schema_store.as_ref().ok_or_else(|| {
440            ArrowError::ParseError("Writer schema store required for raw Avro".into())
441        })?;
442        let fingerprints = store.fingerprints();
443        if fingerprints.is_empty() {
444            return Err(ArrowError::ParseError(
445                "Writer schema store must contain at least one schema".into(),
446            ));
447        }
448        let start_fingerprint = self
449            .active_fingerprint
450            .or_else(|| fingerprints.first().copied())
451            .ok_or_else(|| {
452                ArrowError::ParseError("Could not determine initial schema fingerprint".into())
453            })?;
454        let mut cache = IndexMap::with_capacity(fingerprints.len().saturating_sub(1));
455        let mut active_decoder: Option<RecordDecoder> = None;
456        for fingerprint in store.fingerprints() {
457            let avro_schema = match store.lookup(&fingerprint) {
458                Some(schema) => schema,
459                None => {
460                    return Err(ArrowError::ComputeError(format!(
461                        "Fingerprint {fingerprint:?} not found in schema store",
462                    )));
463                }
464            };
465            let writer_schema = avro_schema.schema()?;
466            let record_decoder =
467                self.make_record_decoder_from_schemas(&writer_schema, reader_schema)?;
468            if fingerprint == start_fingerprint {
469                active_decoder = Some(record_decoder);
470            } else {
471                cache.insert(fingerprint, record_decoder);
472            }
473        }
474        let active_decoder = active_decoder.ok_or_else(|| {
475            ArrowError::ComputeError(format!(
476                "Initial fingerprint {start_fingerprint:?} not found in schema store"
477            ))
478        })?;
479        Ok(self.make_decoder_with_parts(
480            active_decoder,
481            Some(start_fingerprint),
482            cache,
483            store.fingerprint_algorithm(),
484        ))
485    }

572/// A high-level Avro `Reader` that reads container-file blocks
573/// and feeds them into a row-level [`Decoder`].
574#[derive(Debug)]
575pub struct Reader<R: BufRead> {
576    reader: R,
577    header: Header,
578    decoder: Decoder,
579    block_decoder: BlockDecoder,
580    block_data: Vec<u8>,
581    block_count: usize,
582    block_cursor: usize,
583    finished: bool,
584}
585
586impl<R: BufRead> Reader<R> {
587    /// Return the Arrow schema discovered from the Avro file header
588    pub fn schema(&self) -> SchemaRef {
589        self.decoder.schema()
590    }
591
592    /// Return the Avro container-file header
593    pub fn avro_header(&self) -> &Header {
594        &self.header
595    }
596
597    /// Reads the next [`RecordBatch`] from the Avro file or `Ok(None)` on EOF
598    fn read(&mut self) -> Result<Option<RecordBatch>, ArrowError> {
599        'outer: while !self.finished && !self.decoder.batch_is_full() {
600            while self.block_cursor == self.block_data.len() {
601                let buf = self.reader.fill_buf()?;
602                if buf.is_empty() {
603                    self.finished = true;
604                    break 'outer;
605                }
606                // Try to decode another block from the buffered reader.
607                let consumed = self.block_decoder.decode(buf)?;
608                self.reader.consume(consumed);
609                if let Some(block) = self.block_decoder.flush() {
610                    // Successfully decoded a block.
611                    self.block_data = if let Some(ref codec) = self.header.compression()? {
612                        codec.decompress(&block.data)?
613                    } else {
614                        block.data
615                    };
616                    self.block_count = block.count;
617                    self.block_cursor = 0;
618                } else if consumed == 0 {
619                    // The block decoder made no progress on a non-empty buffer.
620                    return Err(ArrowError::ParseError(
621                        "Could not decode next Avro block from partial data".to_string(),
622                    ));
623                }
624            }
625            // Decode as many rows as will fit in the current batch
626            if self.block_cursor < self.block_data.len() {
627                let (consumed, records_decoded) = self
628                    .decoder
629                    .decode_block(&self.block_data[self.block_cursor..], self.block_count)?;
630                self.block_cursor += consumed;
631                self.block_count -= records_decoded;
632            }
633        }
634        self.decoder.flush_block()
635    }
636}

639    type Item = Result<RecordBatch, ArrowError>;
640
641    fn next(&mut self) -> Option<Self::Item> {
642        self.read().transpose()
643    }

709    fn decode_stream<S: Stream<Item = Bytes> + Unpin>(
710        mut decoder: Decoder,
711        mut input: S,
712    ) -> impl Stream<Item = Result<RecordBatch, ArrowError>> {
713        async_stream::try_stream! {
714            if let Some(data) = input.next().await {
715                let consumed = decoder.decode(&data)?;
716                if consumed < data.len() {
717                    Err(ArrowError::ParseError(
718                        "did not consume all bytes".to_string(),
719                    ))?;
720                }
721            }
722            if let Some(batch) = decoder.flush()? {
723                yield batch
724            }
725        }
726    }

784    fn encode_zigzag(value: i64) -> Vec<u8> {
785        let mut n = ((value << 1) ^ (value >> 63)) as u64;
786        let mut out = Vec::new();
787        loop {
788            if (n & !0x7F) == 0 {
789                out.push(n as u8);
790                break;
791            } else {
792                out.push(((n & 0x7F) | 0x80) as u8);
793                n >>= 7;
794            }
795        }
796        out
797    }

800        let mut msg = make_prefix(fp);
801        msg.extend_from_slice(&encode_zigzag(value));
802        msg
803    }
